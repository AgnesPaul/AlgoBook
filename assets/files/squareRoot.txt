Given an integer x, find it’s square root. If x is not a perfect square, then return floor(√x).

def floorSqrt(x): 
    if (x == 0 or x == 1): 
        return x 
    i = 1; result = 1
    while (result <= x): 
        i += 1
        result = i * i  
    return i - 1
  
#Driver function
x = 11
print(floorSqrt(x))
